#include <iostream>
using namespace std;

int main() 
{   
    cout << "size of data types:" << endl;
    cout << "\nSize of char: " << sizeof(char) << " bytes" << endl;
    cout << "\nSize of int: " << sizeof(int) << " bytes" << endl;
    cout << "\nSize of float: " << sizeof(float) << " bytes" << endl;
    cout << "\nSize of double: " << sizeof(double) << " bytes" << endl;

    return 0;
}
